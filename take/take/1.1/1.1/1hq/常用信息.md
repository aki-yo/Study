```java
private Map<String, List<String>> getJxdhUserMapList() {  
    Map<String,List<String>> result = new HashMap<>();  
    try {  
        //接口获取  
        JSONObject obj = (JSONObject) httpService.get(jxRoleUrl, Collections.emptyMap(), JSONObject.class);  
        if (!obj.isEmpty()) {  
            JSONObject codes = obj.getJSONObject("data");  
            if (!codes.isEmpty()) {  
                //核心代表  
                JSONArray core = codes.getJSONArray("core");  
                if (!core.isEmpty()) {  
                    List<String> list = core.toJavaList(String.class);  
                    result.put("core", list);  
                }  
                //LPDT  
                JSONArray lpdt = codes.getJSONArray("lpdt");  
                if (!lpdt.isEmpty()) {  
                    List<String> list = lpdt.toJavaList(String.class);  
                    result.put("lpdt", list);  
                }  
                //数据源owner及赋权人  
                JSONArray dataOwner = codes.getJSONArray("dataOwner");  
                if (!dataOwner.isEmpty()) {  
                    List<String> list = dataOwner.toJavaList(String.class);  
                    result.put("dataOwner", list);  
                }  
            }  
        }  
    } catch (Exception e) {  
        log.info("同步绩效人员error", e);  
    }  
    return result;  
}
```

```
        SELECT 
            U1.C_HID,
```
